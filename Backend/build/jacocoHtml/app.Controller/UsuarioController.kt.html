package app.Controller



import app.domain.*
import app.serializers.View


import io.swagger.v3.oas.annotations.Operation
import org.springframework.web.bind.annotation.*
import app.service.UsuarioService
import com.fasterxml.jackson.annotation.JsonView
import org.springframework.beans.factory.annotation.Autowired


@RestController
@CrossOrigin(origins=["*"])
class UsuarioController {

    @Autowired
    lateinit var usuarioService: UsuarioService

    @PostMapping("/usuario/login")
    @Operation(summary ="Devuelve el usuario cuyo username y contraseÃ±a coincide con lo que pasamos como parÃ¡metro")
    @JsonView(View.Usuario.DatosBasicos::class)
    fun buscarLoguearse(@RequestBody usuario: Usuario): Usuario = usuarioService.buscar(usuario)

    @GetMapping("/usuarios/{id}")
    @Operation(summary = "Devuelve usuario por id")
    fun traerUsuario(@PathVariable id: Long) = usuarioService.getUsuario(id)

   @PutMapping("/actualizarUsuario/{id}")
    @Operation(summary ="Permite actualizar la informaciÃ³n basica de un usuario")
    fun actualizarUsuario(@PathVariable id: Long, @RequestBody usuarioBody: Usuario): Usuario =
        usuarioService.actualizarInfoUsuario(id, usuarioBody)

    @PutMapping("/usuarioCompar/{id}")
    @Operation(summary ="Permite actualizar las compras de un usuario")
    fun agregarCompra(@PathVariable id:Long, @RequestBody nuevoItems: MutableCollection<Item>): Usuario =
        usuarioService.actualizarComprasUsuario(id, nuevoItems)

}
