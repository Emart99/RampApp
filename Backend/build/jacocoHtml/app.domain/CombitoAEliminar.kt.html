package app.domain

class CombitoAEliminar{
    val productosIntegrantes: MutableCollection <Producto> = mutableListOf()
    val ADICIONALXPRODUCTO = 20
    val DESCUENTOXLOTE= 0.15
    val loteDisponible: MutableCollection <Lote> = mutableListOf()
    private var id = 0

    fun precio():Double{
        return productosIntegrantes.fold(0.00) { acum, producto -> acum + (producto.precio() + ADICIONALXPRODUCTO) } * DESCUENTOXLOTE
    }

    fun setId(idAsignar: Int) {
        id = idAsignar
    }

    fun getId(): Int {
        return id
    }

    fun agregarLote(unLote: Lote){
        loteDisponible.add(unLote)
    }

    fun agregarProducto(unProducto: Producto){
        productosIntegrantes.add(unProducto)
    }

    fun actualizarse(nuevoCombo:Combo) {
        val combo = nuevoCombo as CombitoAEliminar
    }



}
