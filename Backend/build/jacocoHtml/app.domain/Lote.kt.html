package app.domain

import java.time.LocalDate


class Lote {
    var idLote = 0
    var fechaIngresoDeposito: LocalDate = LocalDate.now()
    var unidadesDisponibles: Int = 0

    fun modificarLote(prodCarrito: ProdCarrito){
            if(validacionLotes(prodCarrito)){
                this.unidadesDisponibles = this.unidadesDisponibles - prodCarrito.cantidad
            }
    }

    fun validacionLotes(prodCarrito: ProdCarrito): Boolean{
        if (this.idLote == prodCarrito.lote){
            return this.unidadesDisponibles >= prodCarrito.cantidad
        }
        return false

    }

    fun estaVacio(): Boolean {
        return  this.unidadesDisponibles == 0
    }
}
