package app.domain

interface TipoProducto {

    fun adicionalPrecio(precioBase: Int): Double{
        return 0.00
    }
}

class Piso(var tipo: TipoPiso, var medida: String = "", var acabado: TipoAcabado) : TipoProducto {

    override fun adicionalPrecio(precioBase: Int): Double{
        if(tipo == TipoPiso.AltoTransito) return precioBase*0.2
        else return 0.00
    }
}

class Pintura(var litros: Double = 0.00, var rendimiento: Int = 0, var color: String ="") : TipoProducto {

    override fun adicionalPrecio(precioBase: Int): Double{
        if(rendimiento > 8) return precioBase*0.25
        else return 0.00
    }
}
